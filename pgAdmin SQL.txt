CREATE TABLE CLAIMS_DATA(
CLAIM_UID TEXT PRIMARY KEY NOT NULL,
CORD_UID TEXT NOT NULL,
ASOFDT DATE  NULL,
CREATE_DT TIMESTAMP NULL,
CREATE_USER_ID TEXT NULL,
FOREIGN KEY (CORD_UID)
  REFERENCES CORD_DATA(CORD_UID)
);

CREATE TABLE CORD_DATA (
CORD_UID TEXT PRIMARY KEY NOT NULL,
TITLE TEXT NULL,
DOI TEXT NULL,
PUBLISH_TIME DATE NULL,            
JOURNAL TEXT NULL,
COUNTRY TEXT NULL,
INSTITUTION TEXT NULL,
AUTHORS_COUNT INT NULL
);
